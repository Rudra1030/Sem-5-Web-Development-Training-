JavaScript Loop Examples

For Loop

// Loops from i = 0 to i < 5, incrementing i by 1 each time
for (let i = 0; i < 5; i++) {
    console.log("For Loop iteration: " + i);
}

While Loop

// Initializes counter before loop
let j = 0;
// Loops while condition j < 5 is true
while (j < 5) {
    console.log("While Loop iteration: " + j);
    // Increment counter to avoid infinite loop
    j++;
}

Do-While Loop

// Executes loop body first, then checks condition
let k = 0;
do {
    console.log("Do-While Loop iteration: " + k);
    k++;
} while (k < 5);

For...in Loop

// Iterates over enumerable properties (keys) of an object
const person = {name: "Alice", age: 30, city: "New York"};
for (let key in person) {
    console.log(key + ": " + person[key]);
}

For...of Loop

// Iterates over iterable objects like arrays
const fruits = ["apple", "banana", "cherry"];
for (let fruit of fruits) {
    console.log("Fruit: " + fruit);
}
